class Solution {
public:
    int minPathSum(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        vector<vector<int>> f(m, vector<int>(n));
        for (int i = 0; i < m; i ++) 
            for(int j = 0; j < n; j ++) {
                if (i == 0 && j == 0) {
                    f[0][0] = grid[0][0];
                } else if (i == 0) {
                    f[i][j] = f[i][j - 1] + grid[i][j];
                } else if (j == 0) {
                    f[i][j] = f[i - 1][j] + grid[i][j];
                } else {
                    f[i][j] = min(f[i - 1][j], f[i][j - 1]) + grid[i][j];
                }
            }
        
        return f[m - 1][n - 1];
    }
};

class Solution {
public:
    int minPathSum(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        vector<int> f(n);
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (i == 0 && j == 0) {
                    f[0] = grid[0][0];
                } else if (i == 0) {
                    f[j] = f[j - 1] + grid[i][j];
                } else if (j == 0) {
                    f[j] = f[j] + grid[i][j];  // f[j] ä¸ºä¸Šä¸€è¡Œçš„ f[j]

            }
